/*-----------------------------------------------

{

	{

		switch(cmd)
		{
			case 'Q':

				break;

				break;

				head = addData(head);
				break;
			case 'D':
				head = delData(head);

			case 'C':
				head = changeData(head);

				{
					fileSave(head, filename);

				}
				else
					head = fileSave(head, filename);
				break;
			default:
				continue;
		}
	} // while for program

}
